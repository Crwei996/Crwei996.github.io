#include<bits/stdc++.h>
using namespace std;
int a[10090],b[10090],c[10090];
char s1[10090],s2[10090];
int i,la,lb,lc;
int main(){
scanf(“%s”,s1);
scanf(“%s”,s2);
la=strlen(s1);
lb=strlen(s2);
for(i=0;i<la;i++){
a[i]=s1[la-i-1]-‘0’;
}
for(i=0;i<lb;i++){
b[i]=s2[lb-i-1]-‘0’;
}
if(la>lb){
lc=la;
}else{
lc=lb;
}
lc+=1;
for(i=0;i<lc;i++){
c[i]+=a[i]+b[i];
c[i+1]=c[i]/10;
c[i]=c[i]%10;
}
if(c[lc-1]==0&&lc>1){
lc–;
}
for(i=lc-1;i>=0;i–){
printf(“%d”,c[i]);
}
printf(“\n”);
return 0;
}

int query(int x){//费用计算
int fate=0,sum,timeday=0,endtimeday=0,z,m;
// int Month[12]={31,28,31,30,31,30,31,31,30,31,30,31};
if(room[x].method==1){//按天计算
timeday+=(time[x].year-2020)*365;
for(z=0;z<x-1;z++){//假如是某个月多少号就加到前一个月
timeday+=Month[z];
}
timeday+=time[x].day;
endtimeday+=(endtime[x].Syear-2020)*365;
for(m=0;m<x-1;m++){
endtimeday+=Month[m];
}
endtimeday+=endtime[x].Sday; //结束时间计算 刚刚
if(endtime[x].Shour>12){ //只要第二天的时间过了十二点就算多住一天了
endtimeday++;
}//按小时计算
if(room[x].kind==1){
if(room[x].levle==1){//豪华双人间
sum=(endtimeday-timeday)*Luxurious_double_Hour;
}else{//标准双人间
sum=(endtimeday-timeday)*Standard_double_Hour;
}
}else{
if(room[x].levle==1){//豪华单人间
sum=(endtimeday-timeday)*Luxurious_single_Hour;
}else{//标准单人间
sum=(endtimeday-timeday)*Standard_single_Hour;
}
}
return sum;
}else{
timeday+=(time[x].year-2020)36524;
for(z=0;z<x-1;z++){
timeday+=(Month[z]*24);
}
timeday+=time[x].hour;
endtimeday+=(endtime[x].Syear-2020)36524;
for(m=0;m<x-1;m++){
endtimeday+=(Month[m]*24);
}
endtimeday+=endtime[x].Shour;
if(room[x].kind==1){
if(room[x].levle==1){//豪华双人间
sum=(endtimeday-timeday)*Luxurious_double_HourQ;
}else{//标准双人间
sum=(endtimeday-timeday)*Standard_double_HourQ;
}
}else{
if(room[x].levle==1){//豪华单人间
sum=(endtimeday-timeday)*Luxurious_single_HourQ;
}else{//标准单人间
sum=(endtimeday-timeday)*Standard_single_HourQ;
}
}
return sum;
}
