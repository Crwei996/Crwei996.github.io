高精度加法

#include<bits/stdc++.h>
using namespace std;
int a[10090],b[10090],c[10090];
char s1[10090],s2[10090];
int i,la,lb,lc;
int main(){
scanf(“%s”,s1);
scanf(“%s”,s2);
la=strlen(s1);
lb=strlen(s2);
for(i=0;i<la;i++){
a[i]=s1[la-i-1]-‘0’;
}
for(i=0;i<lb;i++){
b[i]=s2[lb-i-1]-‘0’;
}
if(la>lb){
lc=la;
}else{
lc=lb;
}
lc+=1;
for(i=0;i<lc;i++){
c[i]+=a[i]+b[i];
c[i+1]=c[i]/10;
c[i]=c[i]%10;
}
if(c[lc-1]==0&&lc>1){
lc–;
}
for(i=lc-1;i>=0;i–){
printf(“%d”,c[i]);
}
printf(“\n”);
return 0;
}
